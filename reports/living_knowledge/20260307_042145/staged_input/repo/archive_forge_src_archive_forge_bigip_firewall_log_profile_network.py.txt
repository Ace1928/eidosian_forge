    def _validate_aggregate_rate(self, aggregate_rate):
        if aggregate_rate is None:
            return None
        if aggregate_rate == 'indefinite':
            return 4294967295

            return int(aggregate_rate)
        raise F5ModuleError(
            "Valid 'maximum_age' must be in range 0 - 4294967295, or 'indefinite'."

    @classmethod
    def load(cls, filename: str) -> "KANWrapper":
        """
        Load a network's weights, biases, and learnable parameters from a file.

        Args:
            filename (str): The file path to load the network from.
                            # Abbreviation: filename
                            # Description: The path to the file from which the network's state will be loaded.
                            # Typical Value: A string representing a valid file path, e.g., "model.pth".
                            # Range: Any valid file path as a string.

        Returns:
            KANWrapper: An instance of the KANWrapper class with loaded weights, biases, and learnable parameters.
        """
        try:
            with open(filename, "rb") as file:
                state = torch.load(file)
            instance = cls(
                width=state["learnable_params"]["width"],  # Abbreviation: width
                # Description: The width parameter of the network.
                # Typical Value: An integer representing the width, e.g., 128.
                # Range: Any valid integer value.
                grid=state["learnable_params"]["grid"],  # Abbreviation: grid
                # Description: The grid parameter of the network.
                # Typical Value: A list or array representing the grid, e.g., [1, 2, 3].
                # Range: Any valid list or array.
                k=state["learnable_params"]["k"],  # Abbreviation: k
                # Description: The k parameter of the network.
                # Typical Value: An integer representing the k value, e.g., 3.
                # Range: Any valid integer value.
            )
            instance.load_state_dict(state["model_state_dict"])
            instance.optimizer.load_state_dict(state["optimizer_state_dict"])
            instance.learnable_params = state["learnable_params"]
            logging.info(f"Model loaded from {filename}")
            return instance
        except Exception as e:
            logging.error(f"Error loading model: {e}")
            raise

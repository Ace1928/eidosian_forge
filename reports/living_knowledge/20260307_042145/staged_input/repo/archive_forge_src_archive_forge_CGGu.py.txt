def log_function_call(func):
    """
    Decorator to log function calls.
    Purpose:
        This decorator logs the entry and exit of a function, providing insights into the function's usage
        and aiding in debugging and monitoring.
    """

    def wrapper(*args, **kwargs):
        logging.info(f"Entering {func.__name__}")
        result = func(*args, **kwargs)
        logging.info(f"Exiting {func.__name__}")
        return result

    return wrapper


@log_function_call
def get_historical_data_benzinga_api(
    symbol: str, start_date: str, interval: str
) -> pd.DataFrame:
    """
    Fetches historical stock data for a given symbol from a specified start date at a given interval using the Benzinga API.

    This function sends a GET request to the Benzinga API with the specified parameters and processes the returned JSON data into a pandas DataFrame.
    The DataFrame structure is aligned with the expected format for further analysis and visualization within the trading bot.

    Parameters:
    - symbol (str): The stock symbol to fetch historical data for.
    - start_date (str): The start date for the historical data in YYYY-MM-DD format.
    - interval (str): The interval for the historical data (e.g., "1d" for daily).

    Returns:
    - pd.DataFrame: A DataFrame containing the historical stock data with columns for open, high, low, close, and volume.

    Raises:
    - Exception: If the request to the API fails or the response data structure is unexpected.

    Example:
    >>> get_historical_data("AAPL", "1993-01-01", "1W")
    """

    # Define the URL for the API endpoint
    url = "https://api.benzinga.com/api/v2/bars"
    # Parameters to be sent with the HTTP request
    querystring = {
        "token": "YOUR API KEY",
        "symbols": symbol,
        "from": start_date,
        "interval": interval,
    }

    # Initialize an empty DataFrame to hold the historical data
    df = pd.DataFrame()

    try:
        # Sending a GET request to the API
        response = requests.get(url, params=querystring)
        response.raise_for_status()  # Raises an HTTPError if the response was an unsuccessful status code

        # Parsing the JSON response
        hist_json = response.json()

        # Check if the response contains the expected data structure
        if hist_json and isinstance(hist_json, list) and "candles" in hist_json[0]:
            # Converting the JSON data to a pandas DataFrame
            df = pd.DataFrame(hist_json[0]["candles"])
        else:
            raise ValueError("Unexpected JSON structure received from the API.")

    except requests.exceptions.HTTPError as http_err:
        logging.error(
            f"HTTP error occurred while fetching historical data for {symbol}: {http_err}"
        )
        raise
    except ValueError as val_err:
        logging.error(f"Value error: {val_err}")
        raise
    except Exception as e:
        logging.error(f"Failed to fetch historical data for {symbol}: {e}")
        raise

    # Ensure the DataFrame contains the expected columns
    expected_columns = ["open", "high", "low", "close", "volume"]
    if not all(column in df.columns for column in expected_columns):
        raise ValueError(
            f"DataFrame missing one or more expected columns: {expected_columns}"
        )

    return df

@log_function_call
def get_historical_data_google_finance(
    credentials_path: str,
    ticker: str = "NASDAQ:AAPL",
    start_date: str = "2003-01-01",
    end_date: str = "2023-01-01",
) -> pd.DataFrame:
    """
    Fetches historical stock data for a given ticker from a specified start date to end date using Google Finance data in Google Sheets.

    This function uses the gspread library to access Google Sheets and fetch historical stock data that is populated using the GOOGLEFINANCE
    function. It requires a valid credentials JSON file for authentication. The data is returned as a pandas DataFrame with columns for date
    and close price. The DataFrame structure is aligned with the expected format for further analysis and visualization within the trading bot.

    Parameters:
    - credentials_path (str): The path to the JSON file containing the Google Sheets API credentials.
    - ticker (str): The stock ticker symbol to fetch historical data for. Default is "NASDAQ:AAPL".
    - start_date (str): The start date for the historical data in YYYY-MM-DD format. Default is "2003-01-01".
    - end_date (str): The end date for the historical data in YYYY-MM-DD format. Default is "2023-01-01".

    Returns:
    - pd.DataFrame: A DataFrame containing the historical stock data with columns for date and close price.

    Raises:
    - Exception: If there is an error accessing Google Sheets or parsing the data.

    Example:
    >>> get_historical_data_google_finance("path/to/credentials.json", "NASDAQ:GOOGL", "2020-01-01", "2021-01-01")
    """

    try:
        # Authorize the Google Sheets client using the provided credentials
        client = gspread.authorize(
            OAuth2Credentials.from_json_keyfile_name(credentials_path)
        )

        # Open the Google Sheet containing the stock data
        sheet_title = f"{ticker.replace(':', '_')} Stock Data"
        sheet = client.open(sheet_title).sheet1

        # Get all values from the sheet as a list of lists
        data = sheet.get_all_values()

        # Convert the data to a pandas DataFrame
        df = pd.DataFrame(data[1:], columns=data[0])

        # Convert the 'Date' column to datetime format
        df["Date"] = pd.to_datetime(df["Date"])

        # Filter the data based on the specified start and end dates
        mask = (df["Date"] >= start_date) & (df["Date"] <= end_date)
        df = df.loc[mask]

        # Reset the index
        df.reset_index(drop=True, inplace=True)

        return df

    except Exception as e:
        logging.error(f"Failed to fetch historical data for {ticker}: {e}")
        raise


# Function to create and setup the Google Sheet
def setup_google_sheets(
    credentials_path: str,
    ticker: str = "NASDAQ:AAPL",
    start_date: str = "2003-01-01",
    end_date: str = "2023-01-01",
) -> str:
    """
    Creates a new Google Sheet and populates it with historical stock data using the GOOGLEFINANCE function.

    This function uses the gspread library to create a new Google Sheet and populate it with historical stock data
    for a given ticker using the GOOGLEFINANCE function. It requires a valid credentials JSON file for authentication.
    The function returns the URL of the newly created Google Sheet.

    Parameters:
    - credentials_path (str): The path to the JSON file containing the Google Sheets API credentials.
    - ticker (str): The stock ticker symbol to fetch historical data for. Default is "NASDAQ:AAPL".
    - start_date (str): The start date for the historical data in YYYY-MM-DD format. Default is "2003-01-01".
    - end_date (str): The end date for the historical data in YYYY-MM-DD format. Default is "2023-01-01".

    Returns:
    - str: The URL of the newly created Google Sheet containing the historical stock data.

    Raises:
    - Exception: If there is an error creating or populating the Google Sheet.

    Example:
    >>> setup_google_sheets("path/to/credentials.json", "NASDAQ:GOOGL", "2020-01-01", "2021-01-01")
    """

    try:
        # Authorize the Google Sheets client using the provided credentials
        client = gspread.authorize(
            OAuth2Credentials.from_json_keyfile_name(credentials_path)
        )

        # Create a new Google Sheet
        sheet_title = f"{ticker.replace(':', '_')} Stock Data"
        sheet = client.create(sheet_title)

        # Get the first worksheet in the sheet
        worksheet = sheet.get_worksheet(0)

        # Construct the GOOGLEFINANCE function with the specified parameters
        finance_formula = f'=GOOGLEFINANCE("{ticker}", "close", DATE({start_date}), DATE({end_date}), "DAILY")'

        # Write the GOOGLEFINANCE function to cell A1 in the worksheet
        worksheet.update("A1", finance_formula)

        return sheet.url

    except Exception as e:
        logging.error(f"Failed to setup Google Sheet for {ticker}: {e}")
        raise

def implement_strategy(aapl: pd.DataFrame, investment: float) -> None:
    """
    Implements a trading strategy based on Donchian Channels and prints the results.

    Parameters:
    - aapl (pd.DataFrame): The DataFrame containing AAPL stock data with Donchian Channels.
    - investment (float): The initial investment amount.

    Returns:
    - None

    Example:
    >>> implement_strategy(aapl_df, 10000)
    """
    in_position = False  # Flag to track if currently holding a position
    equity = investment  # Current equity
    no_of_shares = 0  # Number of shares bought

    try:
        for i in range(3, len(aapl)):
            # Buying condition: stock's high equals the upper Donchian Channel and not currently in position
            if aapl["high"][i] == aapl["dcu"][i] and not in_position:
                no_of_shares = math.floor(equity / aapl.close[i])
                equity -= no_of_shares * aapl.close[i]
                in_position = True
                logging.info(
                    f"BUY: {no_of_shares} Shares are bought at ${aapl.close[i]} on {str(aapl.index[i])[:10]}"
                )
            # Selling condition: stock's low equals the lower Donchian Channel and currently in position
            elif aapl["low"][i] == aapl["dcl"][i] and in_position:
                equity += no_of_shares * aapl.close[i]
                in_position = False
                logging.info(
                    f"SELL: {no_of_shares} Shares are sold at ${aapl.close[i]} on {str(aapl.index[i])[:10]}"
                )
        # Closing position at the end if still in position
        if in_position:
            equity += no_of_shares * aapl.close[i]
            logging.info(
                f"\nClosing position at {aapl.close[i]} on {str(aapl.index[i])[:10]}"
            )
            in_position = False

        # Calculating earnings and ROI
        earning = round(equity - investment, 2)
        roi = round(earning / investment * 100, 2)
        logging.info(f"EARNING: ${earning} ; ROI: {roi}%")
    except Exception as e:
        logging.error(f"Error implementing strategy: {e}")
        raise

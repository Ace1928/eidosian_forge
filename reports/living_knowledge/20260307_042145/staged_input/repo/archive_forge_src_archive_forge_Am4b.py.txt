"""

class AdvancedLogger:
    def __init__(
        self,
        log_folder: str,
        log_file: str,
        logger_name: str = "advanced_logger",
        level: int = logging.INFO,
    ):
        self.log_folder = log_folder
        self.log_file = os.path.join(log_folder, log_file)
        self.logger_name = logger_name
        self.level = level
        self.logger = self._setup_logger()

    def _setup_logger(self) -> logging.Logger:
        # Ensure the log directory exists
        os.makedirs(self.log_folder, exist_ok=True)

        # Setup advanced logging with a rotating file handler
        log_formatter = logging.Formatter(
            "%(asctime)s - %(levelname)s - %(message)s", datefmt="%Y-%m-%d %H:%M:%S"
        )
        log_handler = RotatingFileHandler(
            self.log_file,
            mode="a",
            maxBytes=10 * 1024 * 1024,
            backupCount=10,
            encoding="utf-8",
            delay=False,
        )
        log_handler.setFormatter(log_formatter)

        logger = logging.getLogger(self.logger_name)
        logger.setLevel(self.level)
        logger.addHandler(log_handler)

        return logger

    def get_logger(self) -> logging.Logger:
        return self.logger

"""
This section of the program handles chunking and parsing of text data from various file formats, ensuring all information is extracted verbatim.
"""


class TextExtractor:
    def __init__(self, file_path: str):
        self.file_path = file_path

    def extract_text_chunks(self) -> List[str]:
        _, file_extension = os.path.splitext(self.file_path)
        text_chunks = []

        try:
            if file_extension == ".md":
                text_chunks = self._extract_md()
            elif file_extension == ".txt":
                text_chunks = self._extract_txt()
            elif file_extension == ".docx":
                text_chunks = self._extract_docx()
            elif file_extension == ".pdf":
                text_chunks = self._extract_pdf()
            elif file_extension == ".csv":
                text_chunks = self._extract_csv()
            elif file_extension == ".json":
                text_chunks = self._extract_json()
            elif file_extension == ".xml":
                text_chunks = self._extract_xml()
            else:
                logging.warning(f"Unsupported file format: {file_extension}")
        except Exception as e:
            logging.error(f"Error processing file: {self.file_path}", exc_info=True)
            raise RuntimeError(
                f"Failed to process file {self.file_path} due to an error: {str(e)}"
            )

        return text_chunks

    def _extract_md(self) -> List[str]:
        with open(self.file_path, "r", encoding="utf-8") as file:
            content = file.read()
            html_content = markdown.markdown(content)
            soup = BeautifulSoup(html_content, "html.parser")
            return [
                p.get_text()
                for p in soup.find_all(["p", "h1", "h2", "h3", "h4", "h5", "h6", "li"])
            ]

    def _extract_txt(self) -> List[str]:
        with open(self.file_path, "r", encoding="utf-8") as file:
            content = file.read()
            return content.split("\n\n")

    def _extract_docx(self) -> List[str]:
        content = docx2txt.process(self.file_path)
        return content.split("\n\n")

    def _extract_pdf(self) -> List[str]:
        with pdfplumber.open(self.file_path) as pdf:
            return [page.extract_text() for page in pdf.pages]

    def _extract_csv(self) -> List[str]:
        with open(self.file_path, "r", encoding="utf-8") as file:
            reader = csv.reader(file)
            return ["\n".join(row) for row in reader]

    def _extract_json(self) -> List[str]:
        with open(self.file_path, "r", encoding="utf-8") as file:
            data = json.load(file)
            return [json.dumps(data, indent=2)]

    def _extract_xml(self) -> List[str]:
        with open(self.file_path, "r", encoding="utf-8") as file:
            data = xmltodict.parse(file.read())
            return [json.dumps(data, indent=2)]


"""
This section of the program handles the processing of files, whether it's a single file or a directory.
"""


class UniversalFileProcessor:
    def __init__(self, path: str):
        self.path = path

    def process_files(self) -> List[str]:
        """
        Process files in the given path, whether it's a single file or a directory.

        Returns:
            List[str]: A list of text chunks extracted from the files.
        """
        all_text_chunks = []

        if os.path.isfile(self.path):
            all_text_chunks.extend(self._process_single_file(self.path))
        elif os.path.isdir(self.path):
            for root, _, files in os.walk(self.path):
                for file in files:
                    file_path = os.path.join(root, file)
                    all_text_chunks.extend(self._process_single_file(file_path))
        else:
            logging.error(f"Invalid path: {self.path}")

        return all_text_chunks

    def _process_single_file(self, file_path: str) -> List[str]:
        """
        Process a single file and extract text chunks.

        Args:
            file_path (str): The path to the file to be processed.

        Returns:
            List[str]: A list of text chunks extracted from the file.
        """
        try:
            extractor = TextExtractor(file_path)
            return extractor.extract_text_chunks()
        except Exception as e:
            logging.error(f"Error processing file {file_path}: {e}", exc_info=True)
            return []
